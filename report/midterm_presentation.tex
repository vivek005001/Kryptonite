% Problem Statement (PS)
% ---------------------------------------------------
\begin{frame}{Problem Statement}
  \textbf{Selected PS:} Automated Mobile Security Vulnerability Assessment
  
  \vspace{0.5cm}
  
  \textbf{Why this problem?}
  \begin{itemize}
    \item Mobile ecosystem (iOS/Android) handles critical user data.
    \item Manual penetration testing is slow and requires specialized skills.
    \item Developers often overlook security best practices (hardcoded keys, weak crypto).
    \item Need for a unified tool to scan both IPA and APK files.
  \end{itemize}
  
  \vspace{0.5cm}
  \textbf{Goal:} "Kryptonite" â€” A seamless CLI tool to automate static analysis and detect vulnerabilities early in the dev cycle.
\end{frame}

% ---------------------------------------------------
% Approach
% ---------------------------------------------------
\begin{frame}{Our Approach: Modular Architecture}
  \begin{center}
    % Architecture Mindmap
    \includegraphics[width=0.85\textwidth]{kryptonite_architecture_mindmap_1771051712399.png}
  \end{center}
  
  \vspace{0.2cm}
  \textbf{Key Components:}
  \begin{enumerate}
    \item \textbf{Platform-Agnostic Core:} Normalized `AppContext` for unified data handling.
    \item \textbf{Specialized Parsers:} `apk_parser` (using struct unpacking) and `ipa_parser` (zip/plist).
    \item \textbf{Extensible Analyzers:} Easy-to-add python modules for new checks.
  \end{enumerate}
\end{frame}
